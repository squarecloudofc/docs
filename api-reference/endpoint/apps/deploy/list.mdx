title: "List Deployments"
sidebarTitle: "[...]/deployments"
description: "Get the last 10 deployments of an application from the last 24 hours."
api: "GET https:/api.squarecloud.app/v2/apps/{app_id}/deployments"
authMethod: "none"
---

<ParamField header="Authorization" type="string" placeholder="API KEY" required>
  The API key for your account. You can find this in your [account settings](https://squarecloud.app/account).
</ParamField>

<ParamField path="app_id" type="string" placeholder="Application ID" required>
  The ID of the application. You can find this in the URL of your application's dashboard.
</ParamField>

### Response

<ResponseField name="status" type="string">
  Indicates whether the call was successful. "success" if successful, "error" if not.
</ResponseField>

<ResponseField name="response" type="array">

The contents of the response.

<Expandable title="Toggle object">

<ResponseField name="id" type="string">
  The ID of the deploy.
</ResponseField>

<ResponseField name="state" type="string">
  The state of the deploy. Can be "pending", "clone", "success", or "error".
</ResponseField>

<ResponseField name="date" type="string">
  The date the deploy was created.
</ResponseField>

</Expandable>
</ResponseField>
